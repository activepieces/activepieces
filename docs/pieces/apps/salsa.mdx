      <Card title="Get supporters " >
        Get all supporters after date

        <Expandable title="Properties">
          
<ResponseField name="API Url" required type="Short_text">
API url acquired from your Salsa API
</ResponseField>
            
<ResponseField name="A list of ids to search for"  type="Array">
Filters results based on the ID for the supporters
</ResponseField>
            
<ResponseField name="Identifier Type"  type="Dropdown">
Select the object
</ResponseField>
            
<ResponseField name="The string to search for"  type="Short_text">
The value to search for if identifierType is SEARCH_STRING. If you provide 2 words separated by a space, it will search first and last names respectively. A single search term will search first name, last name and email
</ResponseField>
            
<ResponseField name="Starting Modified Date" required type="Short_text">
Starting date time to base a search
</ResponseField>
            
<ResponseField name="Last Modified Date"  type="Short_text">
End date time for a search. Optional and if not provided defaults to now
</ResponseField>
            
<ResponseField name="The starting index of items"  type="Number">
Starting index to retrieve values from. This allows you to paginate larger results set
</ResponseField>
            
<ResponseField name="The count of items"  type="Number">
The maximum number of items to retrieve.
</ResponseField>
            
<ResponseField name="Include cell-phone"  type="Checkbox">
Should cell-phone only supporters be included in the results
</ResponseField>
            
<ResponseField name="Include Normal"  type="Checkbox">
Should normal supporters be included in the results
</ResponseField>
